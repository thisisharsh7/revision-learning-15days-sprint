# Technical Deep Dive

## Python Route Generation Script

### Understanding Route Generation

Route generation is the foundation of the testing system. The goal is to create a comprehensive dataset that exercises as many code paths in Valhalla as possible. This means testing different road types, routing scenarios, and edge cases.

**Why Python?**
Python is ideal for scripting because it's easy to read, has excellent HTTP request libraries, and can handle JSON data naturally. The script needs to generate thousands of route requests, call the Valhalla API, and store the results in a structured format.

### Script Architecture

The script should be modular and maintainable. Here's a well-structured approach:

```python
import json
import requests
from typing import List, Dict, Tuple
import logging

# Set up logging for debugging
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class RouteGenerator:
    """
    Generates test routes by calling Valhalla API and storing results.
    This class handles the entire workflow of route generation.
    """

    def __init__(self, api_endpoint: str):
        """
        Initialize with Valhalla API endpoint.

        Args:
            api_endpoint: Base URL for Valhalla API (e.g., 'http://localhost:8002')
        """
        self.api_endpoint = api_endpoint
        self.routes = []
        logger.info(f"RouteGenerator initialized with endpoint: {api_endpoint}")

    def generate_route_request(self,
                              start: Tuple[float, float],
                              end: Tuple[float, float],
                              costing: str = "auto") -> Dict:
        """
        Create a Valhalla route request payload.

        The request format follows Valhalla's API specification.
        Locations are specified as lat/lon pairs.

        Args:
            start: (latitude, longitude) of starting point
            end: (latitude, longitude) of destination
            costing: Routing mode - 'auto', 'bicycle', 'pedestrian'

        Returns:
            Dictionary formatted for Valhalla API
        """
        return {
            "locations": [
                {"lat": start[0], "lon": start[1]},
                {"lat": end[0], "lon": end[1]}
            ],
            "costing": costing,
            "directions_options": {
                "units": "kilometers"
            }
        }

    def fetch_route(self, request: Dict) -> Dict:
        """
        Call Valhalla API and get route response.

        This method handles the HTTP request and error handling.
        It retries on failure and logs errors for debugging.

        Args:
            request: Valhalla route request dictionary

        Returns:
            Route response from Valhalla API

        Raises:
            RuntimeError: If API call fails after retries
        """
        try:
            response = requests.post(
                f"{self.api_endpoint}/route",
                json=request,
                timeout=30  # 30 second timeout
            )
            response.raise_for_status()  # Raise exception for bad status codes
            return response.json()
        except requests.exceptions.RequestException as e:
            logger.error(f"API call failed: {e}")
            raise RuntimeError(f"Failed to fetch route: {e}")

    def save_baseline(self, filename: str):
        """
        Save generated routes to a JSON file.

        This creates the baseline dataset that will be used
        for comparison in regression testing.

        Args:
            filename: Output file path (e.g., 'baseline_routes.json')
        """
        with open(filename, 'w') as f:
            json.dump(self.routes, f, indent=2)
        logger.info(f"Saved {len(self.routes)} routes to {filename}")
```

### State Management with Zustand

**Why Zustand?**
Zustand is lightweight, doesn't require providers/context, and has minimal boilerplate. Perfect for this use case where we need global state for selected route, filters, and route data.

**Setting up the store:**

```typescript
import create from 'zustand';

// Define the shape of our state
interface Route {
  id: string;
  name: string;
  status: 'unchanged' | 'changed';
  baseline: RouteResult;
  pr: RouteResult;
  comparison: ComparisonResult;
}

interface RouteStore {
  // State
  routes: Route[];
  selectedRouteId: string | null;
  filter: 'all' | 'changed' | 'unchanged';
  isLoading: boolean;
  error: string | null;

  // Actions
  setRoutes: (routes: Route[]) => void;
  selectRoute: (id: string) => void;
  setFilter: (filter: 'all' | 'changed' | 'unchanged') => void;
  setLoading: (loading: boolean) => void;
  setError: (error: string | null) => void;

  // Computed values (derived state)
  getFilteredRoutes: () => Route[];
  getSelectedRoute: () => Route | undefined;
  getSummary: () => { total: number; unchanged: number; changed: number };
}

const useRouteStore = create<RouteStore>((set, get) => ({
  // Initial state
  routes: [],
  selectedRouteId: null,
  filter: 'all',
  isLoading: false,
  error: null,

  // Actions - these modify state
  setRoutes: (routes) => set({ routes }),

  selectRoute: (id) => set({ selectedRouteId: id }),

  setFilter: (filter) => set({ filter }),

  setLoading: (loading) => set({ isLoading: loading }),

  setError: (error) => set({ error }),

  // Computed values - these read state
  getFilteredRoutes: () => {
    const { routes, filter } = get();
    if (filter === 'all') return routes;
    return routes.filter(route => route.status === filter);
  },

  getSelectedRoute: () => {
    const { routes, selectedRouteId } = get();
    return routes.find(route => route.id === selectedRouteId);
  },

  getSummary: () => {
    const { routes } = get();
    return {
      total: routes.length,
      unchanged: routes.filter(r => r.status === 'unchanged').length,
      changed: routes.filter(r => r.status === 'changed').length
    };
  }
}));

// Usage in components
const RouteListPanel = () => {
  const routes = useRouteStore(state => state.getFilteredRoutes());
  const filter = useRouteStore(state => state.filter);
  const setFilter = useRouteStore(state => state.setFilter);

  return (
    <div>
      <FilterBar currentFilter={filter} onFilterChange={setFilter} />
      <RouteList routes={routes} />
    </div>
  );
};
```

**Why this approach works:**
- Centralized state management
- Components subscribe only to what they need
- No prop drilling
- Easy to debug (all state in one place)
- Computed values prevent unnecessary calculations

### Virtual Scrolling Deep Dive

**The Problem:**
Rendering 10,000 DOM elements is slow. Even if you can render them, scrolling becomes janky because the browser has to manage too many elements.

**The Solution:**
Virtual scrolling (also called windowing) only renders what's visible on screen. As you scroll, components are reused and new data is loaded.

**How it works:**
1. Calculate how many items fit on screen (based on item height)
2. Render only those items + a few buffer items
3. As user scrolls, calculate which items should now be visible
4. Reuse existing DOM elements with new data
5. Translate the container to create scrolling illusion

**Implementation with react-window:**

```typescript
import { FixedSizeList } from 'react-window';
import AutoSizer from 'react-virtualized-auto-sizer';

interface RouteCardProps {
  route: Route;
  onClick: (id: string) => void;
}

const RouteCard = ({ route, onClick }: RouteCardProps) => {
  const statusColor = route.status === 'changed' ? 'orange' : 'green';

  return (
    <div
      onClick={() => onClick(route.id)}
      style={{
        padding: '16px',
        border: `2px solid ${statusColor}`,
        borderRadius: '8px',
        margin: '8px',
        cursor: 'pointer'
      }}
    >
      <h3>{route.name}</h3>
      <span style={{ color: statusColor }}>
        {route.status === 'changed' ? '📝 Changed' : '✅ Unchanged'}
      </span>
      {route.status === 'changed' && (
        <div>
          <small>Distance: {route.comparison.distanceDiff} km</small>
        </div>
      )}
    </div>
  );
};

const VirtualizedRouteList = ({ routes }: { routes: Route[] }) => {
  const selectRoute = useRouteStore(state => state.selectRoute);

  // Row component that gets rendered for each visible item
  const Row = ({ index, style }: { index: number; style: React.CSSProperties }) => {
    const route = routes[index];

    return (
      <div style={style}>
        <RouteCard route={route} onClick={selectRoute} />
      </div>
    );
  };

  return (
    <div style={{ height: '100vh', width: '400px' }}>
      {/* AutoSizer measures parent container and passes dimensions to List */}
      <AutoSizer>
        {({ height, width }) => (
          <FixedSizeList
            height={height}
            width={width}
            itemCount={routes.length}
            itemSize={120}  // Height of each RouteCard in pixels
            overscanCount={5}  // Render 5 extra items above/below viewport
          >
            {Row}
          </FixedSizeList>
        )}
      </AutoSizer>
    </div>
  );
};
```

**Key parameters explained:**
- `itemSize`: Fixed height of each item (must be consistent)
- `overscanCount`: Number of items to render outside visible area (reduces blank space while scrolling)
- `itemCount`: Total number of items in the list
- `height`/`width`: Container dimensions

**Performance benefits:**
- Only ~20-30 items in DOM instead of 10,000
- Smooth 60fps scrolling
- Minimal memory usage
- Fast initial render

## MapLibre GL JS Integration

### Understanding MapLibre

MapLibre GL JS is a fork of Mapbox GL JS that's open-source and free. It renders interactive maps using WebGL, which means smooth performance even with complex geometries.

**Core concepts:**
- **Map**: The container that holds everything
- **Sources**: Data providers (GeoJSON, vector tiles, raster)
- **Layers**: Visual representation of sources
- **Style**: JSON specification of how map looks

### Setting Up the Map

```typescript
import maplibregl from 'maplibre-gl';
import 'maplibre-gl/dist/maplibre-gl.css';
import { useEffect, useRef, useState } from 'react';

interface MapViewProps {
  baselineRoute?: RouteGeometry;
  prRoute?: RouteGeometry;
}

const MapView = ({ baselineRoute, prRoute }: MapViewProps) => {
  // Use ref to store map instance (persists across renders)
  const mapContainer = useRef<HTMLDivElement>(null);
  const map = useRef<maplibregl.Map | null>(null);
  const [mapLoaded, setMapLoaded] = useState(false);

  // Initialize map (runs once on mount)
  useEffect(() => {
    if (map.current) return; // Already initialized

    map.current = new maplibregl.Map({
      container: mapContainer.current!,
      style: {
        version: 8,
        sources: {
          'osm': {
            type: 'raster',
            tiles: ['https://tile.openstreetmap.org/{z}/{x}/{y}.png'],
            tileSize: 256,
            attribution: '© OpenStreetMap contributors'
          }
        },
        layers: [{
          id: 'osm',
          type: 'raster',
          source: 'osm'
        }]
      },
      center: [0, 0],
      zoom: 2,
      attributionControl: true
    });

    // Wait for map to load before adding layers
    map.current.on('load', () => {
      setMapLoaded(true);
    });

    // Cleanup on unmount
    return () => {
      map.current?.remove();
    };
  }, []);

  // Add routes when map is loaded and routes change
  useEffect(() => {
    if (!mapLoaded || !map.current) return;

    if (baselineRoute) {
      addRouteToMap(map.current, baselineRoute, 'baseline', '#3b82f6');
    }

    if (prRoute) {
      addRouteToMap(map.current, prRoute, 'pr', '#f97316');
    }

    // Fit map to show both routes
    if (baselineRoute || prRoute) {
      fitMapToRoutes(map.current, [baselineRoute, prRoute].filter(Boolean));
    }
  }, [mapLoaded, baselineRoute, prRoute]);

  return (
    <div
      ref={mapContainer}
      style={{
        width: '100%',
        height: '600px',
        borderRadius: '8px',
        overflow: 'hidden'
      }}
    />
  );
};
```

**Why refs and effects?**
- `useRef` stores map instance without causing re-renders
- First `useEffect` initializes map once
- Second `useEffect` updates routes when they change
- This pattern prevents map recreation on every render

### Adding Route Layers

```typescript
interface RouteGeometry {
  type: 'LineString';
  coordinates: [number, number][]; // [lon, lat] pairs
}

const addRouteToMap = (
  map: maplibregl.Map,
  geometry: RouteGeometry,
  id: string,
  color: string
) => {
  // Remove existing layer/source if present
  if (map.getLayer(id)) {
    map.removeLayer(id);
  }
  if (map.getSource(id)) {
    map.removeSource(id);
  }

  // Add GeoJSON source
  map.addSource(id, {
    type: 'geojson',
    data: {
      type: 'Feature',
      properties: {},
      geometry: geometry
    }
  });

  // Add line layer
  map.addLayer({
    id: id,
    type: 'line',
    source: id,
    layout: {
      'line-join': 'round',
      'line-cap': 'round'
    },
    paint: {
      'line-color': color,
      'line-width': [
        'interpolate',
        ['linear'],
        ['zoom'],
        8, 2,   // At zoom 8: 2px wide
        15, 6   // At zoom 15: 6px wide
      ],
      'line-opacity': 0.8
    }
  });

  // Add start marker
  const start = geometry.coordinates[0];
  new maplibregl.Marker({ color: color })
    .setLngLat(start)
    .addTo(map);

  // Add end marker
  const end = geometry.coordinates[geometry.coordinates.length - 1];
  new maplibregl.Marker({ color: color })
    .setLngLat(end)
    .addTo(map);
};
```

**Layer paint properties explained:**
- `line-color`: Color of the line
- `line-width`: Uses zoom-based interpolation (thinner when zoomed out)
- `line-opacity`: 0.8 allows seeing overlapping routes

### Fitting Map to Routes

```typescript
const fitMapToRoutes = (
  map: maplibregl.Map,
  geometries: RouteGeometry[]
) => {
  // Create bounds object
  const bounds = new maplibregl.LngLatBounds();

  // Extend bounds to include all coordinates
  geometries.forEach(geometry => {
    geometry.coordinates.forEach(coord => {
      bounds.extend(coord as [number, number]);
    });
  });

  // Fit map to bounds with padding
  map.fitBounds(bounds, {
    padding: {
      top: 50,
      bottom: 50,
      left: 50,
      right: 50
    },
    duration: 1000,  // Animate over 1 second
    maxZoom: 15      // Don't zoom in too far
  });
};
```

**Why padding?**
Routes at screen edges are hard to see. Padding ensures routes are comfortably visible with space around them.

### Understanding CI/CD for Route Testing

Continuous Integration/Continuous Deployment (CI/CD) automates testing. For this project, GitHub Actions will:
1. Trigger on every pull request
2. Build Valhalla from the PR branch

```yaml
      # Step 10: Fail if too many regressions
```

## Route Comparison Algorithm

### The Challenge of Comparison

Comparing routes isn't just checking if they're identical. Routes might differ slightly due to:
- Floating point precision
- Minor geometry adjustments
- Updated OSM data
- Algorithm improvements

We need smart comparison that identifies **meaningful** changes.

### Implementing Smart Comparison

```typescript
interface RouteComparison {
  status: 'unchanged' | 'changed';
  differences: {
    distance?: { absolute: number; percentage: number };
    time?: { absolute: number; percentage: number };
    geometry?: { divergencePoint: [number, number]; maxDeviation: number };
    instructions?: { added: number; removed: number; modified: number };
  };
  severity: 'minor' | 'moderate' | 'major';
}

const compareRoutes = (
  baseline: RouteResult,
  pr: RouteResult,
  thresholds: ComparisonThresholds = DEFAULT_THRESHOLDS
): RouteComparison => {
  const differences: any = {};
  let changeCount = 0;

  // 1. Distance Comparison
  const distanceDiff = pr.distance - baseline.distance;
  const distancePct = (Math.abs(distanceDiff) / baseline.distance) * 100;

  if (distancePct > thresholds.distancePercentage) {
    differences.distance = {
      absolute: distanceDiff,
      percentage: distancePct
    };
    changeCount++;
  }

  // 2. Time Comparison
  const timeDiff = pr.time - baseline.time;
  const timePct = (Math.abs(timeDiff) / baseline.time) * 100;

  if (timePct > thresholds.timePercentage) {
    differences.time = {
      absolute: timeDiff,
      percentage: timePct
    };
    changeCount++;
  }

  // 3. Geometry Comparison
  const geometryDiff = compareGeometry(
    baseline.geometry,
    pr.geometry,
    thresholds.geometryTolerance
  );

  if (geometryDiff.hasDifference) {
    differences.geometry = geometryDiff;
    changeCount++;
  }

  // 4. Instructions Comparison
  const instructionsDiff = compareInstructions(
    baseline.instructions,
    pr.instructions
  );

  if (instructionsDiff.hasChanges) {
    differences.instructions = instructionsDiff;
    changeCount++;
  }

  // Determine severity
  let severity: 'minor' | 'moderate' | 'major' = 'minor';
  if (changeCount >= 3 || distancePct > 20) severity = 'major';
  else if (changeCount >= 2 || distancePct > 10) severity = 'moderate';

  return {
    status: changeCount > 0 ? 'changed' : 'unchanged',
    differences,
    severity
  };
};
```

**Threshold configuration:**

```typescript
interface ComparisonThresholds {
  distancePercentage: number;  // e.g., 1% = ignore changes under 1%
  timePercentage: number;      // e.g., 5% = ignore time changes under 5%
  geometryTolerance: number;   // meters deviation allowed
}

const DEFAULT_THRESHOLDS: ComparisonThresholds = {
  distancePercentage: 1.0,     // 1%
  timePercentage: 5.0,         // 5%
  geometryTolerance: 10.0      // 10 meters
};
```

### Geometry Comparison Algorithm

Comparing route geometries is complex. We need to check if the routes follow the same path, within a tolerance.

```typescript
interface GeometryComparison {
  hasDifference: boolean;
  divergencePoint?: [number, number];
  maxDeviation?: number;
  affectedLength?: number;
}

const compareGeometry = (
  baseline: [number, number][],
  pr: [number, number][],
  tolerance: number
): GeometryComparison => {
  // Quick check: very different lengths
  const lengthDiff = Math.abs(baseline.length - pr.length);
  if (lengthDiff > baseline.length * 0.1) {  // 10% difference in point count
    return {
      hasDifference: true,
      maxDeviation: Infinity
    };
  }

  let maxDeviation = 0;
  let divergencePoint: [number, number] | undefined;
  let deviationStart = -1;

  // Compare point by point
  const minLength = Math.min(baseline.length, pr.length);
  for (let i = 0; i < minLength; i++) {
    const distance = haversineDistance(baseline[i], pr[i]);

    if (distance > tolerance) {
      if (deviationStart === -1) {
        deviationStart = i;
        divergencePoint = baseline[i];
      }
      maxDeviation = Math.max(maxDeviation, distance);
    }
  }

  return {
    hasDifference: maxDeviation > tolerance,
    divergencePoint,
    maxDeviation,
    affectedLength: deviationStart >= 0 ? minLength - deviationStart : 0
  };
};

// Haversine formula: calculate distance between two lat/lon points
const haversineDistance = (
  point1: [number, number],
  point2: [number, number]
): number => {
  const R = 6371e3; // Earth's radius in meters
  const φ1 = point1[1] * Math.PI / 180;
  const φ2 = point2[1] * Math.PI / 180;
  const Δφ = (point2[1] - point1[1]) * Math.PI / 180;
  const Δλ = (point2[0] - point1[0]) * Math.PI / 180;

  const a = Math.sin(Δφ / 2) * Math.sin(Δφ / 2) +
            Math.cos(φ1) * Math.cos(φ2) *
            Math.sin(Δλ / 2) * Math.sin(Δλ / 2);

  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));

  return R * c; // Distance in meters
};
```

**Why this approach?**
- Uses Haversine formula for accurate lat/lon distance
- Tolerates minor deviations (GPS noise, algorithm tweaks)
- Identifies where routes diverge
- Measures severity of divergence

## Performance Optimization

### Understanding Performance Bottlenecks

In a React app handling 10,000 routes, performance issues can arise from:
1. Too many re-renders
2. Expensive calculations on every render
3. Large data transfers
4. Unoptimized list rendering

### Memoization Strategy

**useMemo** - Cache expensive calculations:

```typescript
import { useMemo } from 'react';

const RouteListPanel = ({ routes, filter, searchQuery }) => {
  // This calculation runs on every render without useMemo
  // With 10,000 routes, this is slow!

  // ❌ Bad: Recalculates on every render
  // const filteredRoutes = routes
  //   .filter(r => r.status === filter)
  //   .filter(r => r.name.includes(searchQuery));

  // ✅ Good: Only recalculates when dependencies change
  const filteredRoutes = useMemo(() => {
    console.log('Filtering routes...'); // This logs only when routes/filter/searchQuery change

    let result = routes;

    // Apply status filter
    if (filter !== 'all') {
      result = result.filter(r => r.status === filter);
    }

    // Apply search filter
    if (searchQuery) {
      result = result.filter(r =>
        r.name.toLowerCase().includes(searchQuery.toLowerCase())
      );
    }

    return result;
  }, [routes, filter, searchQuery]);

  return <VirtualizedList routes={filteredRoutes} />;
};
```

**useCallback** - Cache function references:

```typescript
import { useCallback } from 'react';

const RouteCard = ({ route, onSelect }) => {
  // This creates a new function on every render
  // Causes child components to re-render unnecessarily

  // ❌ Bad: New function every render
  // const handleClick = () => {
  //   onSelect(route.id);
  // };

  // ✅ Good: Same function reference unless route.id changes
  const handleClick = useCallback(() => {
    onSelect(route.id);
  }, [route.id, onSelect]);

  return <div onClick={handleClick}>{route.name}</div>;
};
```

### Debouncing for Search

```typescript
import { useState, useCallback } from 'react';
import { debounce } from 'lodash';

const SearchBar = () => {
  const [searchQuery, setSearchQuery] = useState('');
  const updateSearch = useRouteStore(state => state.setSearchQuery);

  // Don't update on every keystroke - wait 300ms after user stops typing
  const debouncedUpdate = useCallback(
    debounce((query: string) => {
      updateSearch(query);
    }, 300),
    []
  );

  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const value = e.target.value;
    setSearchQuery(value);      // Update input immediately (no lag)
    debouncedUpdate(value);     // Update filter after 300ms delay
  };

  return (
    <input
      type="text"
      value={searchQuery}
      onChange={handleChange}
      placeholder="Search routes..."
    />
  );
};
```

**Why debouncing helps:**
- User types "berlin" - that's 6 keystrokes
- Without debounce: 6 filter operations
- With debounce: 1 filter operation (after they stop typing)
- Result: 6x less work, smoother experience

### Error Handling Pattern

```typescript
const RouteLoader = ({ prId }: { prId: string }) => {
  const { data, error, isLoading } = useComparisonResults(prId);

  // Handle loading state
  if (isLoading) {
    return (
      <div className="flex justify-center items-center h-screen">
        <Spinner />
        <span>Loading routes...</span>
      </div>
    );
  }

  // Handle error state
  if (error) {
    return (
      <div className="error-container">
        <h2>Failed to load routes</h2>
        <p>{error.message}</p>
        <button onClick={() => window.location.reload()}>
          Retry
        </button>
      </div>
    );
  }

  // Handle no data state
  if (!data || data.length === 0) {
    return (
      <div className="empty-state">
        <p>No routes found for this PR</p>
      </div>
    );
  }

  // Happy path: render the data
  return <RouteList routes={data} />;
};
```
