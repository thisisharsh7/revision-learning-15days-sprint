## Python Route Generation Script

### Script Structure

```python
import json
import requests
from typing import List, Dict, Tuple

class RouteGenerator:
    def __init__(self, api_endpoint: str):
        self.api_endpoint = api_endpoint
        self.routes = []

    def generate_route_request(self, start: Tuple[float, float],
                                end: Tuple[float, float],
                                costing: str = "auto") -> Dict:
        return {
            "locations": [
                {"lat": start[0], "lon": start[1]},
                {"lat": end[0], "lon": end[1]}
            ],
            "costing": costing
        }

    def fetch_route(self, request: Dict) -> Dict:
        response = requests.post(
            f"{self.api_endpoint}/route",
            json=request
        )
        return response.json()
```

## Route Comparison Algorithm

### Comparison Logic

```typescript
interface RouteComparison {
  status: 'unchanged' | 'changed';
  differences: {
    distance?: number;
    time?: number;
    geometry?: boolean;
    instructions?: boolean;
  };
}

const compareRoutes = (baseline: Route, pr: Route): RouteComparison => {
  const differences: any = {};

  // Distance comparison (threshold: 1%)
  const distanceDiff = Math.abs(baseline.distance - pr.distance);
  const distanceThreshold = baseline.distance * 0.01;
  if (distanceDiff > distanceThreshold) {
    differences.distance = pr.distance - baseline.distance;
  }

  // Time comparison (threshold: 5%)
  const timeDiff = Math.abs(baseline.time - pr.time);
  const timeThreshold = baseline.time * 0.05;
  if (timeDiff > timeThreshold) {
    differences.time = pr.time - baseline.time;
  }

  // Geometry comparison
  if (!geometryEquals(baseline.geometry, pr.geometry)) {
    differences.geometry = true;
  }

  // Instructions comparison
  if (!instructionsEqual(baseline.instructions, pr.instructions)) {
    differences.instructions = true;
  }

  return {
    status: Object.keys(differences).length > 0 ? 'changed' : 'unchanged',
    differences
  };
};
```

### Geometry Comparison

```typescript
const geometryEquals = (geom1: number[][], geom2: number[][]): boolean => {
  if (geom1.length !== geom2.length) return false;

  const tolerance = 0.00001; // ~1 meter

  for (let i = 0; i < geom1.length; i++) {
    const [lon1, lat1] = geom1[i];
    const [lon2, lat2] = geom2[i];

    if (Math.abs(lon1 - lon2) > tolerance ||
        Math.abs(lat1 - lat2) > tolerance) {
      return false;
    }
  }

  return true;
};
```

### Route Interface

```typescript
interface Route {

interface RouteResult {
```

```typescript
const fetchComparisonResults = async (prId: string): Promise<Route[]> => {
  const response = await fetch(
    `https://api.example.com/results/${prId}/comparison.json`
```
